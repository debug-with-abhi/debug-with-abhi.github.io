from flask import Flask, request, render_template_string app = Flask(__name__) HTML = """ Smart Queue Predictor

🤖 Smart Queue Management System

AI-Based Waiting Time Prediction

{% if prediction is not none %}

Prediction Result

People Ahead: {{ people }}

Service Time: {{ service_time }} minutes

Estimated Waiting Time: {{ prediction }} minutes

{% endif %}
""" @app.route("/", methods=["GET", "POST"]) def home(): prediction = None people = None service_time = None if request.method == "POST": people = int(request.form["people"]) service_time = float(request.form["service_time"]) # Waiting time calculation prediction = people * service_time return render_template_string( HTML, prediction=prediction, people=people, service_time=service_time ) if __name__ == "__main__": app.run(debug=True)
